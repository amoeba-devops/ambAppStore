```sql
-- ===================================
-- Migration: 구독관리 알림 만료 기능
-- Date: 2026-03-29
-- ===================================

-- Step 1: plt_subscriptions ENUM에 EXPIRED 추가
ALTER TABLE plt_subscriptions 
  MODIFY COLUMN sub_status 
  ENUM('PENDING','ACTIVE','SUSPENDED','REJECTED','CANCELLED','EXPIRED') 
  NOT NULL DEFAULT 'PENDING';

-- Step 2: plt_notifications 테이블 생성
CREATE TABLE IF NOT EXISTS plt_notifications (
  ntf_id CHAR(36) NOT NULL,
  ent_id CHAR(36) NOT NULL,
  ntf_user_id CHAR(36) NOT NULL,
  ntf_type ENUM(
    'SUB_APPROVED',
    'SUB_REJECTED',
    'SUB_SUSPENDED',
    'SUB_EXPIRED',
    'SUB_EXPIRING_SOON',
    'SYSTEM'
  ) NOT NULL,
  ntf_title VARCHAR(200) NOT NULL,
  ntf_message VARCHAR(500) NOT NULL,
  ntf_ref_type VARCHAR(50) NULL,
  ntf_ref_id CHAR(36) NULL,
  ntf_is_read TINYINT(1) NOT NULL DEFAULT 0,
  ntf_read_at DATETIME(6) NULL,
  ntf_created_at DATETIME(6) NOT NULL DEFAULT CURRENT_TIMESTAMP(6),
  ntf_deleted_at DATETIME(6) NULL,
  PRIMARY KEY (ntf_id),
  INDEX IDX_ntf_user_read (ntf_user_id, ntf_is_read, ntf_created_at DESC),
  INDEX IDX_ntf_ent (ent_id),
  INDEX IDX_ntf_ref (ntf_ref_type, ntf_ref_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;
```

### 5.2 로컬 개발 환경
- `synchronize: true` 설정이므로 TypeORM이 자동으로 스키마 동기화
- `init-db.sql`에 위 SQL 추가하여 Docker 재생성 시에도 적용되도록 함

### 5.3 Rollback SQL

```sql
-- Rollback: plt_notifications 삭제
DROP TABLE IF EXISTS plt_notifications;

-- Rollback: EXPIRED enum 제거 (데이터 없는 경우만 가능)
-- 주의: EXPIRED 상태인 레코드가 있으면 먼저 CANCELLED로 변경 후 실행
UPDATE plt_subscriptions SET sub_status = 'CANCELLED' WHERE sub_status = 'EXPIRED';
ALTER TABLE plt_subscriptions 
  MODIFY COLUMN sub_status 
  ENUM('PENDING','ACTIVE','SUSPENDED','REJECTED','CANCELLED') 
  NOT NULL DEFAULT 'PENDING';
```
